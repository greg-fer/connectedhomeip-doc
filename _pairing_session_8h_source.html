117  void SendStatusReport(Messaging::ExchangeContext * exchangeCtxt, uint16_t protocolCode)
118  {
119  Protocols::SecureChannel::GeneralStatusCode generalCode = (protocolCode == Protocols::SecureChannel::kProtocolCodeSuccess)
120  ? Protocols::SecureChannel::GeneralStatusCode::kSuccess
121  : Protocols::SecureChannel::GeneralStatusCode::kFailure;
122  uint32_t protocolId = Protocols::SecureChannel::Id.ToFullyQualifiedSpecForm();
123 
124  ChipLogDetail(SecureChannel, "Sending status report. Protocol code %d, exchange %d", protocolCode,
125  exchangeCtxt->GetExchangeId());
126 
127  Protocols::SecureChannel::StatusReport statusReport(generalCode, protocolId, protocolCode);
128 
129  Encoding::LittleEndian::PacketBufferWriter bbuf(System::PacketBufferHandle::New(statusReport.Size()));
130  statusReport.WriteToBuffer(bbuf);
131 
132  System::PacketBufferHandle msg = bbuf.Finalize();
133  VerifyOrReturn(!msg.IsNull(), ChipLogError(SecureChannel, "Failed to allocate status report message"));
134 
135  CHIP_ERROR err = exchangeCtxt->SendMessage(Protocols::SecureChannel::MsgType::StatusReport, std::move(msg));
136  if (err != CHIP_NO_ERROR)
137  {
138  ChipLogError(SecureChannel, "Failed to send status report message. %s", ErrorStr(err));
139  }
140  }

142  CHIP_ERROR HandleStatusReport(System::PacketBufferHandle && msg, bool successExpected)
143  {
144  Protocols::SecureChannel::StatusReport report;
145  CHIP_ERROR err = report.Parse(std::move(msg));
146  ReturnErrorOnFailure(err);
147  VerifyOrReturnError(report.GetProtocolId() == Protocols::SecureChannel::Id.ToFullyQualifiedSpecForm(),
148  CHIP_ERROR_INVALID_ARGUMENT);
149 
150  if (report.GetGeneralCode() == Protocols::SecureChannel::GeneralStatusCode::kSuccess &&
151  report.GetProtocolCode() == Protocols::SecureChannel::kProtocolCodeSuccess && successExpected)
152  {
153  OnSuccessStatusReport();
154  }
155  else
156  {
157  err = OnFailureStatusReport(report.GetGeneralCode(), report.GetProtocolCode());
158  }
159 
160  return err;
161  }
