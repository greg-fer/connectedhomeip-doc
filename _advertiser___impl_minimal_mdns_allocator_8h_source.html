37 template <size_t kMaxRecords>
38 class QueryResponderAllocator
39 {
40 public:
41  QueryResponderAllocator()
42  {
43  for (auto & responder : mAllocatedResponders)
44  {
45  responder = nullptr;
46  }
47  for (auto & name : mAllocatedQNameParts)
48  {
49  name = nullptr;
50  }
51  }
52  ~QueryResponderAllocator() { Clear(); }
53 
55  template <typename ResponderType, typename... Args>
56  mdns::Minimal::QueryResponderSettings AddResponder(Args &&... args)
57  {
58  return AddAllocatedResponder(chip::Platform::New<ResponderType>(std::forward<Args>(args)...));
59  }
60 
61  template <typename... Args>
62  mdns::Minimal::FullQName AllocateQName(Args &&... names)
63  {
64  void * storage = AllocateQNameSpace(mdns::Minimal::FlatAllocatedQName::RequiredStorageSize(std::forward<Args>(names)...));
65  if (storage == nullptr)
66  {
67  return mdns::Minimal::FullQName();
68  }
69  return mdns::Minimal::FlatAllocatedQName::Build(storage, std::forward<Args>(names)...);
70  }
71 
72  mdns::Minimal::FullQName AllocateQNameFromArray(char const * const * names, size_t num)
73  {
74  void * storage = AllocateQNameSpace(mdns::Minimal::FlatAllocatedQName::RequiredStorageSizeFromArray(names, num));
75  if (storage == nullptr)
76  {
77  return mdns::Minimal::FullQName();
78  }
79  return mdns::Minimal::FlatAllocatedQName::BuildFromArray(storage, names, num);
80  }
81 
84  void Clear()
85  {
86  // Init clears all responders, so that data can be freed
87  mQueryResponder.Init();
88 
89  // Free all allocated data
90  for (auto & responder : mAllocatedResponders)
91  {
92  if (responder != nullptr)
93  {
94  chip::Platform::Delete(responder);
95  responder = nullptr;
96  }
97  }
98 
99  for (auto & name : mAllocatedQNameParts)
100  {
101  if (name != nullptr)
102  {
103  chip::Platform::MemoryFree(name);
104  name = nullptr;
105  }
106  }
107  }
108  mdns::Minimal::QueryResponder<kMaxRecords + 1> * GetQueryResponder() { return &mQueryResponder; }
109  const mdns::Minimal::RecordResponder * GetResponder(const mdns::Minimal::QType & qtype,
110  const mdns::Minimal::FullQName & qname) const
111  {
112  for (auto & responder : mAllocatedResponders)
113  {
114  if (responder != nullptr && responder->GetQType() == qtype && responder->GetQName() == qname)
115  {
116  return responder;
117  }
118  }
119  return nullptr;
120  }
121  bool IsEmpty() const
122  {
123  for (auto & responder : mAllocatedResponders)
124  {
125  if (responder != nullptr)
126  {
127  return false;
128  }
129  }
130  for (auto & name : mAllocatedQNameParts)
131  {
132  if (name != nullptr)
133  {
134  return false;
135  }
136  }
137  return true;
138  }
139 
140 protected:
141  // For testing.
142  size_t GetMaxAllocatedQNames() { return kMaxAllocatedQNameData; }
143  void * GetQNamePart(size_t idx) { return mAllocatedQNameParts[idx]; }
144  mdns::Minimal::RecordResponder * GetRecordResponder(size_t idx) { return mAllocatedResponders[idx]; }
145 
146 private:
147  static constexpr size_t kMaxAllocatedQNameData = 32;
148  // dynamically allocated items
149  mdns::Minimal::RecordResponder * mAllocatedResponders[kMaxRecords];
150  void * mAllocatedQNameParts[kMaxAllocatedQNameData];
151  // The QueryResponder needs 1 extra space to hold the record for itself.
152  mdns::Minimal::QueryResponder<kMaxRecords + 1> mQueryResponder;
153 
154  mdns::Minimal::QueryResponderSettings AddAllocatedResponder(mdns::Minimal::RecordResponder * newResponder)
155  {
156  if (newResponder == nullptr)
157  {
158  ChipLogError(Discovery, "Responder memory allocation failed");
159  return mdns::Minimal::QueryResponderSettings(); // failed
160  }
161 
162  for (auto & responder : mAllocatedResponders)
163  {
164  if (responder != nullptr)
165  {
166  continue;
167  }
168 
169  responder = newResponder;
170  return mQueryResponder.AddResponder(responder);
171  }
172 
173  Platform::Delete(newResponder);
174  ChipLogError(Discovery, "Failed to find free slot for adding a responder");
175  return mdns::Minimal::QueryResponderSettings();
176  }
177 
178  void * AllocateQNameSpace(size_t size)
179  {
180  for (auto & name : mAllocatedQNameParts)
181  {
182  if (name != nullptr)
183  {
184  continue;
185  }
186 
187  name = chip::Platform::MemoryAlloc(size);
188  if (name == nullptr)
189  {
190  ChipLogError(Discovery, "QName memory allocation failed");
191  }
192  return name;
193  }
194  ChipLogError(Discovery, "Failed to find free slot for adding a qname");
195  return nullptr;
196  }
197 };
