113 constexpr size_t MaxKeyLen(TxtKeyUse use)
114 {
115  size_t max = 0;
116  for (auto & info : Internal::txtFieldInfo)
117  {
118  if (use == info.use)
119  {
120  max = sizeof(info.keyStr) > max ? sizeof(info.keyStr) : max;

124  return max - 1;
