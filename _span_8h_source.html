119 template <class T, size_t N>
120 class FixedSpan
121 {
122 public:
123  using pointer = T *;
124  using const_pointer = const T *;
125 
126  constexpr FixedSpan() : mDataBuf(nullptr) {}
127 
128  // We want to allow construction from things that look like T*, but we want
129  // to make construction from an array use the constructor that asserts the
130  // array is big enough. This requires that both constructors be templates
131  // (because otherwise the non-template would be favored by overload
132  // resolution, since due to decay to pointer it matches just as well as the
133  // template).
134  //
135  // To do that we have a template constructor enabled only when the type
136  // passed to it is a pointer type, and that pointer is to a type that
137  // matches T up to const-ness. Importantly, we do NOT want to allow
138  // subclasses of T here, because they would have a different size and our
139  // Span would not work right.
140  template <
141  class U,
142  typename = std::enable_if_t<std::is_pointer<U>::value &&
143  std::is_same<std::remove_const_t<T>, std::remove_const_t<std::remove_pointer_t<U>>>::value>>
144  constexpr explicit FixedSpan(U databuf) : mDataBuf(databuf)
145  {}
146  template <class U, size_t M, typename = std::enable_if_t<std::is_same<std::remove_const_t<T>, std::remove_const_t<U>>::value>>
147  constexpr explicit FixedSpan(U (&databuf)[M]) : mDataBuf(databuf)
148  {
149  static_assert(M >= N, "Passed-in buffer too small for FixedSpan");
150  }
151 
152  // Allow implicit construction from a FixedSpan of sufficient size over a
153  // type that matches our type, up to const-ness.
154  template <class U, size_t M, typename = std::enable_if_t<std::is_same<std::remove_const_t<T>, std::remove_const_t<U>>::value>>
155  constexpr FixedSpan(FixedSpan<U, M> const & other) : mDataBuf(other.data())
156  {
157  static_assert(M >= N, "Passed-in FixedSpan is smaller than we are");
158  }
159 
160  constexpr pointer data() const { return mDataBuf; }
161  constexpr size_t size() const { return N; }
162  constexpr bool empty() const { return data() == nullptr; }
163 
164  constexpr pointer begin() { return mDataBuf; }
165  constexpr pointer end() { return mDataBuf + N; }
166  constexpr const_pointer begin() const { return mDataBuf; }
167  constexpr const_pointer end() const { return mDataBuf + N; }
168 
169  // Allow data_equal for spans that are over the same type up to const-ness.
170  template <class U, typename = std::enable_if_t<std::is_same<std::remove_const_t<T>, std::remove_const_t<U>>::value>>
171  bool data_equal(const FixedSpan<U, N> & other) const
172  {
173  return (empty() && other.empty()) ||
174  (!empty() && !other.empty() && (memcmp(data(), other.data(), size() * sizeof(T)) == 0));
175  }
176 
177  template <class U, typename = std::enable_if_t<std::is_same<std::remove_const_t<T>, std::remove_const_t<U>>::value>>
178  bool data_equal(const Span<U> & other) const
179  {
180  return (size() == other.size() && (empty() || (memcmp(data(), other.data(), size() * sizeof(T)) == 0)));
181  }
182 
183 private:
184  pointer mDataBuf;
185 };
