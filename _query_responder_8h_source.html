131  bool Accept(Internal::QueryResponderInfo * record) const
132  {
133  if (record->responder == nullptr)
134  {
135  return false;
136  }
137 
138  if (mIncludeAdditionalRepliesOnly && !record->reportNowAsAdditional)
139  {
140  return false;
141  }
142 
143  if ((mIncludeOnlyMulticastBefore > chip::System::Clock::kZero) &&
144  (record->lastMulticastTime >= mIncludeOnlyMulticastBefore))
145  {
146  return false;
147  }
148 
149  if ((mReplyFilter != nullptr) &&
150  !mReplyFilter->Accept(record->responder->GetQType(), record->responder->GetQClass(), record->responder->GetQName()))
151  {
152  return false;
153  }
154  return true;
155  }
156 
157 private:
158  bool mIncludeAdditionalRepliesOnly = false;
159  ReplyFilter * mReplyFilter = nullptr;
160  chip::System::Clock::Timestamp mIncludeOnlyMulticastBefore = chip::System::Clock::kZero;
161 };
