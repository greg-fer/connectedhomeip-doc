97  mTcpSupported = tcpSupported;
98  return *reinterpret_cast<Derived *>(this);
99  }
100  Optional<bool> GetTcpSupported() const { return mTcpSupported; }
101

104  bool mEnableIPv4 = true;

106  size_t mMacLength = 0;
