142  void LogDetail() const
143  {
144 #if CHIP_ENABLE_ROTATING_DEVICE_ID
145  if (rotatingIdLen > 0)
146  {
147  char rotatingIdString[chip::Dnssd::kMaxRotatingIdLen * 2 + 1] = "";
148  Encoding::BytesToUppercaseHexString(rotatingId, rotatingIdLen, rotatingIdString, sizeof(rotatingIdString));
149  ChipLogDetail(Discovery, "Rotating ID: %s", rotatingIdString);
150  }
151 #endif // CHIP_ENABLE_ROTATING_DEVICE_ID
152  if (strlen(deviceName) != 0)
153  {
154  ChipLogDetail(Discovery, "\tDevice Name: %s", deviceName);
155  }
156  if (vendorId > 0)
157  {
158  ChipLogDetail(Discovery, "\tVendor ID: %u", vendorId);
159  }
160  if (productId > 0)
161  {
162  ChipLogDetail(Discovery, "\tProduct ID: %u", productId);
163  }
164  if (deviceType > 0)
165  {
166  ChipLogDetail(Discovery, "\tDevice Type: %u", deviceType);
167  }
168  if (longDiscriminator > 0)
169  {
170  ChipLogDetail(Discovery, "\tLong Discriminator: %u", longDiscriminator);
171  }
172  if (strlen(pairingInstruction) != 0)
173  {
174  ChipLogDetail(Discovery, "\tPairing Instruction: %s", pairingInstruction);
175  }
176  if (pairingHint > 0)
177  {
178  ChipLogDetail(Discovery, "\tPairing Hint: 0x%x", pairingHint);
179  }
180  if (!IsHost(""))
181  {
182  ChipLogDetail(Discovery, "\tHostname: %s", hostName);
183  }
184  for (int j = 0; j < numIPs; j++)
185  {
186 #if CHIP_DETAIL_LOGGING
187  char buf[Inet::IPAddress::kMaxStringLength];
188  char * ipAddressOut = ipAddress[j].ToString(buf);
189  ChipLogDetail(Discovery, "\tIP Address #%d: %s", j + 1, ipAddressOut);
190  (void) ipAddressOut;
191 #endif // CHIP_DETAIL_LOGGING
192  }
193  if (port > 0)
194  {
195  ChipLogDetail(Discovery, "\tPort: %u", port);
196  }
197  ChipLogDetail(Discovery, "\tCommissioning Mode: %u", commissioningMode);
198  }
