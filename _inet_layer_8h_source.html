157  // In some implementations, there may be a shared event / message
158  // queue for the InetLayer used by other system events / messages.
159  //
160  // If the length of that queue is considerably longer than the
161  // number of packet buffers available, it may lead to buffer
162  // exhaustion. As a result, using the queue itself to implement
163  // backpressure is insufficient, and we need an external mechanism
164  // to prevent buffer starvation in the rest of the system and
165  // getting into deadlock situations.
166 
167  // For both UDP and raw network transport traffic we can easily
168  // drop incoming packets without impacting the correctness of
169  // higher level protocols.
170

177 #endif // INET_CONFIG_ENABLE_UDP_ENDPOINT
178  false;
179  }
