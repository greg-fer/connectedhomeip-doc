29 template <typename T>
30 bool IsCurrentInterfaceUsable(T & iterator)
31 {
32  if (!iterator.IsUp() || !iterator.SupportsMulticast())
33  {
34  return false; // not a usable interface
35  }
36  char name[chip::Inet::InterfaceId::kMaxIfNameLength];
37  if (iterator.GetInterfaceName(name, sizeof(name)) != CHIP_NO_ERROR)
38  {
39  ChipLogError(Discovery, "Failed to get interface name.");
40  return false;
41  }
42 
43  // TODO: need a better way to ignore local loopback interfaces/addresses
44  // We do not want to listen on local loopback even though they are up and
45  // support multicast
46  //
47  // Some way to detect 'is local looback' that is smarter (e.g. at least
48  // strict string compare on linux instead of substring) would be better.
49  //
50  // This would reject likely valid interfaces like 'lollipop' or 'lostinspace'
51  if (strncmp(name, "lo", 2) == 0)
52  {
54  return false;
55  }
56  return true;
57 }
