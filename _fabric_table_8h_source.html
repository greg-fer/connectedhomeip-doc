335  ConstFabricIterator(const FabricInfo * start, size_t index, size_t maxSize) : mStart(start), mIndex(index), mMaxSize(maxSize)
336  {
337  if (mIndex >= maxSize)
338  {
339  mIndex = maxSize;
340  }
341  else if (!mStart[mIndex].IsInitialized())
342  {
343  Advance();
344  }
345  }
346  ConstFabricIterator(const ConstFabricIterator &) = default;
347  ConstFabricIterator & operator=(const ConstFabricIterator &) = default;
348 
349  ConstFabricIterator & operator++() { return Advance(); }
350  ConstFabricIterator operator++(int)
351  {
352  ConstFabricIterator other(*this);
353  Advance();
354  return other;
355  }
356 
357  const FabricInfo & operator*() const { return mStart[mIndex]; }
358  const FabricInfo * operator->() const { return mStart + mIndex; }
359 
360  bool operator==(const ConstFabricIterator & other)
361  {
362  if (IsAtEnd())
363  {
364  return other.IsAtEnd();
365  }
366 
367  return (mStart == other.mStart) && (mIndex == other.mIndex) && (mMaxSize == other.mMaxSize);
368  }
369  bool operator!=(const ConstFabricIterator & other) { return !(*this == other); }
370 
371  bool IsAtEnd() const { return (mIndex == mMaxSize); }
372 
373 private:
374  const FabricInfo * mStart;
375  size_t mIndex;
376  size_t mMaxSize;
377 
378  ConstFabricIterator & Advance()
379  {
380  do
381  {
382  if (mIndex < mMaxSize)
383  {
384  mIndex++;
385  }
386  } while (!IsAtEnd() && !mStart[mIndex].IsInitialized());
387 
388  return *this;
389  }
