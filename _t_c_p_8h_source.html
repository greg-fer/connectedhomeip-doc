130 public:
131  TCPBase(ActiveConnectionState * activeConnectionsBuffer, size_t bufferSize, PendingPacket * packetBuffers,
132  size_t packetsBuffersSize) :
133  mActiveConnections(activeConnectionsBuffer),
134  mActiveConnectionsSize(bufferSize), mPendingPackets(packetBuffers), mPendingPacketsSize(packetsBuffersSize)
135  {
136  // activeConnectionsBuffer must be initialized by the caller.
137  for (size_t i = 0; i < mPendingPacketsSize; ++i)
138  {
139  mPendingPackets[i].peerAddress = PeerAddress::Uninitialized();
140  // In the typical case, the TCPBase constructor is invoked from the TCP constructor on its mPendingPackets,
141  // which has not yet been initialized. That means we can't do a normal move assignment or construction of
142  // the PacketBufferHandle, since that would call PacketBuffer::Free on the uninitialized data.
143  new (&mPendingPackets[i].packetBuffer) System::PacketBufferHandle();
144  }
