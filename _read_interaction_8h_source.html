29 /*
30  * A typed read attribute function that takes as input a template parameter that encapsulates the type information
31  * for a given attribute as well as callbacks for success and failure and either returns a decoded cluster-object representation
32  * of the requested attribute through the provided success callback or calls the provided failure callback.
33  *
34  * The AttributeTypeInfo is generally expected to be a ClusterName::Attributes::AttributeName::TypeInfo struct, but any
35  * object that contains type information exposed through a 'DecodableType' type declaration as well as GetClusterId() and
36  * GetAttributeId() methods is expected to work.
37  *
38  */
39 template <typename AttributeTypeInfo>
40 CHIP_ERROR ReadAttribute(Messaging::ExchangeManager * aExchangeMgr, const SessionHandle sessionHandle, EndpointId endpointId,
41  typename TypedReadCallback<AttributeTypeInfo>::OnSuccessCallbackType onSuccessCb,
42  typename TypedReadCallback<AttributeTypeInfo>::OnErrorCallbackType onErrorCb)
43 {
44  app::AttributePathParams attributePath(endpointId, AttributeTypeInfo::GetClusterId(), AttributeTypeInfo::GetAttributeId());
45  app::ReadPrepareParams readParams(sessionHandle);
46  app::ReadClient * readClient = nullptr;
47  app::InteractionModelEngine * engine = app::InteractionModelEngine::GetInstance();
48  CHIP_ERROR err = CHIP_NO_ERROR;
49 
50  readParams.mpAttributePathParamsList = &attributePath;
51  readParams.mAttributePathParamsListSize = 1;
52 
53  auto onDone = [](app::ReadClient * apReadClient, TypedReadCallback<AttributeTypeInfo> * callback) {
54  chip::Platform::Delete(callback);
55  };
56 
57  auto callback = chip::Platform::MakeUnique<TypedReadCallback<AttributeTypeInfo>>(onSuccessCb, onErrorCb, onDone);
58  VerifyOrReturnError(callback != nullptr, CHIP_ERROR_NO_MEMORY);
59 
60  ReturnErrorOnFailure(engine->NewReadClient(&readClient, app::ReadClient::InteractionType::Read, callback.get()));
61 
62  err = readClient->SendReadRequest(readParams);
63  if (err != CHIP_NO_ERROR)
64  {
65  readClient->Shutdown();
66  return err;
67  }
68 
69  //
70  // At this point, we'll get a callback through the OnDone callback above regardless of success or failure
71  // of the read operation to permit us to free up the callback object. So, release ownership of the callback
72  // object now to prevent it from being reclaimed at the end of this scoped block.
73  //
74  callback.release();
75  return err;
76 }
