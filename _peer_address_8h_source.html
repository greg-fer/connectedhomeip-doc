137  void ToString(char * buf, size_t bufSize) const
138  {
139  char ip_addr[kInetMaxAddrLen];
140 
141  char interface[Inet::InterfaceId::kMaxIfNameLength + 1] = {}; // +1 to prepend '%'
142  if (mInterface.IsPresent())
143  {
144  interface[0] = '%';
145  interface[1] = 0;
146  CHIP_ERROR err = mInterface.GetInterfaceName(interface + 1, sizeof(interface) - 1);
147  if (err != CHIP_NO_ERROR)
148  {
149  Platform::CopyString(interface, sizeof(interface), "%(err)");
150  }
151  }
152 
153  switch (mTransportType)
154  {
155  case Type::kUndefined:
156  snprintf(buf, bufSize, "UNDEFINED");
157  break;
158  case Type::kUdp:
159  mIPAddress.ToString(ip_addr);
160 #if INET_CONFIG_ENABLE_IPV4
161  if (mIPAddress.IsIPv4())
162  snprintf(buf, bufSize, "UDP:%s%s:%d", ip_addr, interface, mPort);
163  else
164 #endif
165  snprintf(buf, bufSize, "UDP:[%s%s]:%d", ip_addr, interface, mPort);
166  break;
167  case Type::kTcp:
168  mIPAddress.ToString(ip_addr);
169 #if INET_CONFIG_ENABLE_IPV4
170  if (mIPAddress.IsIPv4())
171  snprintf(buf, bufSize, "TCP:%s%s:%d", ip_addr, interface, mPort);
172  else
173 #endif
174  snprintf(buf, bufSize, "TCP:[%s%s]:%d", ip_addr, interface, mPort);
175  break;
176  case Type::kBle:
177  // Note that BLE does not currently use any specific address.
178  snprintf(buf, bufSize, "BLE");
179  break;
180  default:
181  snprintf(buf, bufSize, "ERROR");
182  break;
183  }
184  }
