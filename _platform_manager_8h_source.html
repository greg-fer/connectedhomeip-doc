313 inline CHIP_ERROR PlatformManager::InitChipStack()
314 {
315  // NOTE: this is NOT thread safe and cannot be as the chip stack lock is prepared by
316  // InitChipStack itself on many platforms.
317  //
318  // In the future, this could be moved into specific platform code (where it can
319  // be made thread safe). In general however, init twice
320  // is likely a logic error and we may want to avoid that path anyway. Likely to
321  // be done once code stabilizes a bit more.
322  if (mInitialized)
323  {
324  return CHIP_NO_ERROR;
325  }
326 
327  CHIP_ERROR err = static_cast<ImplClass *>(this)->_InitChipStack();
328  mInitialized = (err == CHIP_NO_ERROR);
329  return err;
330 }
