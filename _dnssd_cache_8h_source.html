40 template <size_t CACHE_SIZE>
41 class DnssdCache
42 {
43 public:
44  DnssdCache() : elementsUsed(CACHE_SIZE)
45  {
46  for (DnssdCacheEntry & e : mLookupTable)
47  {
48  // each unused entry decrements the count
49  MarkEntryUnused(e);
50  }
51  MdnsLogProgress(Discovery, "construct mdns cache of size %ld", CACHE_SIZE);
52  }
53 
54  // insert this entry into the cache.
55  // return error if cache is full
56  // TODO: have an eviction policy so if the cache is full, an entry may be deleted.
57  // One policy may be Least-time-to-live
58  CHIP_ERROR Insert(PeerId peerId, const Inet::IPAddress & addr, uint16_t port, Inet::InterfaceId iface,
59  System::Clock::Timestamp TTL)
60  {
61  const System::Clock::Timestamp currentTime = mTimeSource.GetMonotonicTimestamp();
62 
63  DnssdCacheEntry * entry;
64 
65  entry = FindPeerId(peerId, currentTime);
66  if (entry)
67  {
68  // update timeout if found entry
69  entry->expiryTime = currentTime + TTL;
70  entry->TTL = TTL; // in case it changes */
71  return CHIP_NO_ERROR;
72  }
73 
74  VerifyOrReturnError(entry = findSlot(currentTime), CHIP_ERROR_TOO_MANY_KEYS);
75 
76  // have a free slot for this entry
77  entry->peerId = peerId;
78  entry->ipAddr = addr;
79  entry->port = port;
80  entry->ifaceId = iface;
81  entry->TTL = TTL;
82  entry->expiryTime = currentTime + TTL;
83  elementsUsed++;
84 
85  return CHIP_NO_ERROR;
86  }
87 
88  CHIP_ERROR Delete(PeerId peerId)
89  {
90  DnssdCacheEntry * pentry;
91  const System::Clock::Timestamp currentTime = mTimeSource.GetMonotonicTimestamp();
92 
93  VerifyOrReturnError(pentry = FindPeerId(peerId, currentTime), CHIP_ERROR_KEY_NOT_FOUND);
94 
95  MarkEntryUnused(*pentry);
96  return CHIP_NO_ERROR;
97  }
98 
99  // given a peerId, find the parameters if its in the cache, or return error
100  CHIP_ERROR Lookup(PeerId peerId, Inet::IPAddress & addr, uint16_t & port, Inet::InterfaceId & iface)
101  {
102  DnssdCacheEntry * pentry;
103  const System::Clock::Timestamp currentTime = mTimeSource.GetMonotonicTimestamp();
104 
105  VerifyOrReturnError(pentry = FindPeerId(peerId, currentTime), CHIP_ERROR_KEY_NOT_FOUND);
106 
107  addr = pentry->ipAddr;
108  port = pentry->port;
109  iface = pentry->ifaceId;
110 
111  return CHIP_NO_ERROR;
112  }
113 
114  // only useful if MDNS_LOGGING is set. If not used, should be optimized out
115  void DumpCache()
116  {
117  int i = 0;
118 
119  MdnsLogProgress(Discovery, "cache size = %d", elementsUsed);
120  for (DnssdCacheEntry & e : mLookupTable)
121  {
122  if (e.peerId == nullPeerId)
123  {
124  MdnsLogProgress(Discovery, "Entry %d unused", i);
125  }
126  else
127  {
128  char address[100];
129 
130  e.ipAddr.ToString(address, sizeof address);
131  MdnsLogProgress(Discovery, "Entry %d: node %lx fabric %lx, port = %d, address = %s", i, e.peerId.GetNodeId(),
132  e.peerId.GetFabricId(), e.port, address);
133  }
134  i++;
135  }
136  }

138 private:
139  struct DnssdCacheEntry
140  {
141  PeerId peerId;
142  Inet::IPAddress ipAddr;
143  uint16_t port;
144  Inet::InterfaceId ifaceId;
145  System::Clock::Timestamp TTL;
146  System::Clock::Timestamp expiryTime;
147  };
148  PeerId nullPeerId; // indicates a cache entry is unused
149  int elementsUsed; // running count of how many entries are used -- for a sanity check
150 
151  DnssdCacheEntry mLookupTable[CACHE_SIZE];
152  Time::TimeSource<Time::Source::kSystem> mTimeSource;
153 
154  DnssdCacheEntry * findSlot(System::Clock::Timestamp currentTime)
155  {
156  for (DnssdCacheEntry & entry : mLookupTable)
157  {
158  if (entry.peerId == nullPeerId)
159  return &entry;
160 
161  if (entry.expiryTime <= currentTime)
162  {
163  MarkEntryUnused(entry);
164  return &entry;
165  }
166  }
167  return nullptr;
168  }
169 
170  DnssdCacheEntry * FindPeerId(PeerId peerId, System::Clock::Timestamp current_time)
171  {
172  for (DnssdCacheEntry & entry : mLookupTable)
173  {
174  if (entry.peerId == peerId)
175  {
176  if (entry.expiryTime < current_time)
177  {
178  MarkEntryUnused(entry);
179  break; // return nullptr
180  }
181  else
182  return &entry;
183  }
184  if (entry.peerId != nullPeerId && entry.expiryTime < current_time)
185  {
186  MarkEntryUnused(entry);
187  }
188  }
189 
190  return nullptr;
191  }
192 
193  // have a method to mark ununused -- so its easy to change
194  void MarkEntryUnused(DnssdCacheEntry & pentry)
195  {
196  pentry.peerId = nullPeerId;
197  elementsUsed--;
198  }
199 };
