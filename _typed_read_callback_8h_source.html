29 /*
30  * This provides an adapter class that implements InteractionModelEngine::InteractionModelDelegate and provides two additional
31  * features:
32  * 1. The ability to pass in std::function closures to permit more flexible programming scenarios than are provided by the strict
33  * delegate interface stipulated by by InteractionModelDelegate.
34  *
35  * 2. Automatic decoding of attribute data provided in the TLVReader by InteractionModelDelegate::OnReportData into a decoded
36  * cluster object.
37  */
38 template <typename AttributeTypeInfo>
39 class TypedReadCallback final : public app::ReadClient::Callback
40 {
41 public:
42  using OnSuccessCallbackType =
43  std::function<void(const app::ConcreteAttributePath & aPath, const typename AttributeTypeInfo::DecodableType & aData)>;
44  using OnErrorCallbackType = std::function<void(const app::ConcreteAttributePath * aPath,
45  Protocols::InteractionModel::Status aIMStatus, CHIP_ERROR aError)>;
46  using OnDoneCallbackType = std::function<void(app::ReadClient * client, TypedReadCallback * callback)>;
47 
48  TypedReadCallback(OnSuccessCallbackType aOnSuccess, OnErrorCallbackType aOnError, OnDoneCallbackType aOnDone) :
49  mOnSuccess(aOnSuccess), mOnError(aOnError), mOnDone(aOnDone)
50  {}
51 
52 private:
53  void OnAttributeData(const app::ReadClient * apReadClient, const app::ConcreteAttributePath & aPath, TLV::TLVReader * apData,
54  const app::StatusIB & status) override
55  {
56  CHIP_ERROR err = CHIP_NO_ERROR;
57  typename AttributeTypeInfo::DecodableType value;
58 
59  VerifyOrExit(status.mStatus == Protocols::InteractionModel::Status::Success, err = CHIP_ERROR_IM_STATUS_CODE_RECEIVED);
60  VerifyOrExit(aPath.mClusterId == AttributeTypeInfo::GetClusterId() &&
61  aPath.mAttributeId == AttributeTypeInfo::GetAttributeId(),
62  CHIP_ERROR_SCHEMA_MISMATCH);
63  VerifyOrExit(apData != nullptr, err = CHIP_ERROR_INVALID_ARGUMENT);
64 
65  err = app::DataModel::Decode(*apData, value);
66  SuccessOrExit(err);
67 
68  mOnSuccess(aPath, value);
69 
70  exit:
71  if (err != CHIP_NO_ERROR)
72  {
73  //
74  // Override status to indicate an error if something bad happened above.
75  //
76  Protocols::InteractionModel::Status imStatus = status.mStatus;
77  if (status.mStatus == Protocols::InteractionModel::Status::Success)
78  {
79  imStatus = Protocols::InteractionModel::Status::Failure;
80  }
81 
82  mOnError(&aPath, imStatus, err);
83  }
84  }
85 
86  void OnError(const app::ReadClient * apReadClient, CHIP_ERROR aError) override
87  {
88  mOnError(nullptr, Protocols::InteractionModel::Status::Failure, aError);
89  }
90 
91  void OnDone(app::ReadClient * apReadClient) override { mOnDone(apReadClient, this); }
92 
93  OnSuccessCallbackType mOnSuccess;
94  OnErrorCallbackType mOnError;
95  OnDoneCallbackType mOnDone;
96 };
