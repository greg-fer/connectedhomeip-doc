30 /*
31  * This provides an adapter class that implements CommandSender::Callback and provides two additional features:
32  * 1. The ability to pass in std::function closures to permit more flexible programming scenarios than are provided by the strict
33  * delegate interface stipulated by CommandSender::Callback
34  *
35  * 2. Automatic decoding of command response data provided in the TLVReader by the CommandSender callback into a decoded cluster
36  * object.
37  */
38 template <typename CommandResponseObjectT>
39 class TypedCommandCallback final : public app::CommandSender::Callback
40 {
41 public:
42  using OnSuccessCallbackType =
43  std::function<void(const app::ConcreteCommandPath &, const app::StatusIB &, const CommandResponseObjectT &)>;
44  using OnErrorCallbackType = std::function<void(const app::StatusIB & aStatus, CHIP_ERROR aError)>;
45  using OnDoneCallbackType = std::function<void(app::CommandSender * commandSender)>;
46 
47  /*
48  * Constructor that takes in success, failure and onDone callbacks.
49  *
50  * The latter can be provided later through the SetOnDoneCallback below in cases where the
51  * TypedCommandCallback object needs to be created first before it can be passed in as a closure
52  * into a hypothetical OnDoneCallback function.
53  */
54  TypedCommandCallback(OnSuccessCallbackType aOnSuccess, OnErrorCallbackType aOnError, OnDoneCallbackType aOnDone = {}) :
55  mOnSuccess(aOnSuccess), mOnError(aOnError), mOnDone(aOnDone)
56  {}
57 
58  void SetOnDoneCallback(OnDoneCallbackType callback) { mOnDone = callback; }
59 
60 private:
61  void OnResponse(app::CommandSender * apCommandSender, const app::ConcreteCommandPath & aCommandPath,
62  const app::StatusIB & aStatus, TLV::TLVReader * aReader) override;
63 
64  void OnError(const app::CommandSender * apCommandSender, const app::StatusIB & aStatus, CHIP_ERROR aError) override
65  {
66  mOnError(aStatus, aError);
67  }
68 
69  void OnDone(app::CommandSender * apCommandSender) override { mOnDone(apCommandSender); }
70 
71  OnSuccessCallbackType mOnSuccess;
72  OnErrorCallbackType mOnError;
73  OnDoneCallbackType mOnDone;
74 };

76 /*
77  * Decodes the data provided by the TLVReader into the templated cluster object that denotes the command response.
78  *
79  * This function specifically decodes command responses that have actual data payloads.
80  */
81 template <typename CommandResponseObjectT>
82 void TypedCommandCallback<CommandResponseObjectT>::OnResponse(app::CommandSender * apCommandSender,
83  const app::ConcreteCommandPath & aCommandPath,
84  const app::StatusIB & aStatus, TLV::TLVReader * aReader)
85 {
86  CommandResponseObjectT response;
87  CHIP_ERROR err = CHIP_NO_ERROR;
88 
89  //
90  // We're expecting response data in this variant of OnResponse. Consequently, aReader should always be
91  // non-null. If it is, it means we received a success status code instead, which is not what was expected.
92  //
93  VerifyOrExit(aReader != nullptr, err = CHIP_ERROR_SCHEMA_MISMATCH);
94 
95  //
96  // Validate that the data response we received matches what we expect in terms of its cluster and command IDs.
97  //
98  VerifyOrExit(aCommandPath.mClusterId == CommandResponseObjectT::GetClusterId() &&
99  aCommandPath.mCommandId == CommandResponseObjectT::GetCommandId(),
100  err = CHIP_ERROR_SCHEMA_MISMATCH);
101 
102  err = app::DataModel::Decode(*aReader, response);
103  SuccessOrExit(err);
104 
105  mOnSuccess(aCommandPath, aStatus, response);
106 
107 exit:
108  if (err != CHIP_NO_ERROR)
109  {
110  app::StatusIB status;
111  status.mStatus = Protocols::InteractionModel::Status::Failure;
112  mOnError(status, err);
113  }
114 }

116 /*
117  * Decodes the data provided by the TLVReader into the templated cluster object that denotes the command response.
118  *
119  * This function specifically decodes command responses that do not have actual data payloads and where the passed in TLVReader
120  * should be null.
121  */
122 template <>
123 inline void TypedCommandCallback<app::DataModel::NullObjectType>::OnResponse(app::CommandSender * apCommandSender,
124  const app::ConcreteCommandPath & aCommandPath,
125  const app::StatusIB & aStatus,
126  TLV::TLVReader * aReader)
127 {
128  //
129  // If we got a valid reader, it means we received response data that we were not expecting to receive.
130  //
131  if (aReader != nullptr)
132  {
133  app::StatusIB status;
134  status.mStatus = Protocols::InteractionModel::Status::Failure;
135  mOnError(status, CHIP_ERROR_SCHEMA_MISMATCH);
136  return;
137  }
138 
139  app::DataModel::NullObjectType nullResp;
140  mOnSuccess(aCommandPath, aStatus, nullResp);
141 }
