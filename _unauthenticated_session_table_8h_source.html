71 /*
72  * @brief
73  * An table which manages UnauthenticatedSessions
74  *
75  * The UnauthenticatedSession entries are rotated using LRU, but entry can be
76  * hold by using UnauthenticatedSessionHandle, which increase the reference
77  * count by 1. If the reference count is not 0, the entry won't be pruned.
78  */
79 template <size_t kMaxSessionCount, Time::Source kTimeSource = Time::Source::kSystem>
80 class UnauthenticatedSessionTable
81 {
82 public:
88  CHECK_RETURN_VALUE
89  Optional<UnauthenticatedSessionHandle> FindOrAllocateEntry(const PeerAddress & address)
90  {
91  UnauthenticatedSession * result = FindEntry(address);
92  if (result != nullptr)
93  return MakeOptional<UnauthenticatedSessionHandle>(*result);
94 
95  CHIP_ERROR err = AllocEntry(address, result);
96  if (err == CHIP_NO_ERROR)
97  {
98  return MakeOptional<UnauthenticatedSessionHandle>(*result);
99  }
100  else
101  {
102  return Optional<UnauthenticatedSessionHandle>::Missing();
103  }
104  }
105 
107  void MarkSessionActive(UnauthenticatedSessionHandle session)
108  {
109  session->SetLastActivityTime(mTimeSource.GetMonotonicTimestamp());
110  }
111 
113  Time::TimeSource<kTimeSource> & GetTimeSource() { return mTimeSource; }
114 
115 private:
122  CHECK_RETURN_VALUE
123  CHIP_ERROR AllocEntry(const PeerAddress & address, UnauthenticatedSession *& entry)
124  {
125  entry = mEntries.CreateObject(address);
126  if (entry != nullptr)
127  return CHIP_NO_ERROR;
128 
129  entry = FindLeastRecentUsedEntry();
130  if (entry == nullptr)
131  {
132  return CHIP_ERROR_NO_MEMORY;
133  }
134 
135  mEntries.ResetObject(entry, address);
136  return CHIP_NO_ERROR;
137  }
138 
144  CHECK_RETURN_VALUE
145  UnauthenticatedSession * FindEntry(const PeerAddress & address)
146  {
147  UnauthenticatedSession * result = nullptr;
148  mEntries.ForEachActiveObject([&](UnauthenticatedSession * entry) {
149  if (MatchPeerAddress(entry->GetPeerAddress(), address))
150  {
151  result = entry;
152  return false;
153  }
154  return true;
155  });
156  return result;
157  }
158 
159  UnauthenticatedSession * FindLeastRecentUsedEntry()
160  {
161  UnauthenticatedSession * result = nullptr;
162  System::Clock::Timestamp oldestTime = System::Clock::Timestamp(std::numeric_limits<System::Clock::Timestamp::rep>::max());
163 
164  mEntries.ForEachActiveObject([&](UnauthenticatedSession * entry) {
165  if (entry->GetReferenceCount() == 0 && entry->GetLastActivityTime() < oldestTime)
166  {
167  result = entry;
168  oldestTime = entry->GetLastActivityTime();
169  }
170  return true;
171  });
172 
173  return result;
174  }

176  // A temporary solution for #11120
177  // Enforce interface match if not null
178  static bool MatchInterface(Inet::InterfaceId i1, Inet::InterfaceId i2)
179  {
180  if (i1.IsPresent() && i2.IsPresent())
181  {
182  return i1 == i2;
183  }
184  else
185  {
186  // One of the interfaces is null.
187  return true;
188  }
189  }
190 
191  static bool MatchPeerAddress(const PeerAddress & a1, const PeerAddress & a2)
192  {
193  if (a1.GetTransportType() != a2.GetTransportType())
194  return false;
195 
196  switch (a1.GetTransportType())
197  {
198  case Transport::Type::kUndefined:
199  return false;
200  case Transport::Type::kUdp:
201  case Transport::Type::kTcp:
202  return a1.GetIPAddress() == a2.GetIPAddress() && a1.GetPort() == a2.GetPort() &&
203  // Enforce interface equal-ness if the address is link-local, otherwise ignore interface
204  // Use MatchInterface for a temporary solution for #11120
205  (a1.GetIPAddress().IsIPv6LinkLocal() ? a1.GetInterface() == a2.GetInterface()
206  : MatchInterface(a1.GetInterface(), a2.GetInterface()));
207  case Transport::Type::kBle:
208  // TODO: complete BLE address comparation
209  return true;
210  }
211 
212  return false;
213  }
