287  /*
288  * The representation of a CHIP_ERROR is structured so that SDK error code constants are small, in order to improve code
289  * density on embedded builds. Arm 32, Xtensa, and RISC-V can all handle 11-bit values in a move-immediate instruction.
290  * Further, SdkPart::kCore is 0 so that the most common errors fit in 8 bits for additional density on some processors.
291  *
292  * 31 28 24 20 16 12 8 4 0 Bit
293  * | | | | | | | | |
294  * | range | value |
295  * | kSdk==0 | 0 |0| part| code | SDK error
296  * | 01 - FF | encapsulated error code | Encapsulated error
297  */
298  static constexpr int kRangeStart = 24;
299  static constexpr int kRangeLength = 8;
300  static constexpr int kValueStart = 0;
301  static constexpr int kValueLength = 24;
302 
303  static constexpr int kSdkPartStart = 8;
304  static constexpr int kSdkPartLength = 3;
305  static constexpr int kSdkCodeStart = 0;
306  static constexpr int kSdkCodeLength = 8;
307 
308  static constexpr StorageType GetField(unsigned int start, unsigned int length, StorageType value)
309  {
310  return (value >> start) & ((1u << length) - 1);
311  }
312  static constexpr StorageType MakeMask(unsigned int start, unsigned int length) { return ((1u << length) - 1) << start; }
313  static constexpr StorageType MakeField(unsigned int start, StorageType value) { return value << start; }
314  static constexpr bool FitsInField(unsigned int length, StorageType value) { return value < (1u << length); }
315 
316  static constexpr StorageType MakeInteger(Range range, StorageType value)
317  {
318  return MakeField(kRangeStart, to_underlying(range)) | MakeField(kValueStart, value);
319  }
320  static constexpr StorageType MakeInteger(SdkPart part, uint8_t code)
321  {
322  return MakeInteger(Range::kSDK, MakeField(kSdkPartStart, to_underlying(part)) | MakeField(kSdkCodeStart, code));
323  }
324  template <unsigned int START, unsigned int LENGTH>
325  struct MaskConstant
326  {
327  static constexpr StorageType value = ((1u << LENGTH) - 1) << START;
328  };
329 
330  // Assert that Range and Value fields fit in StorageType and don't overlap.
331  static_assert(kRangeStart + kRangeLength <= std::numeric_limits<StorageType>::digits, "Range does not fit in StorageType");
332  static_assert(kValueStart + kValueLength <= std::numeric_limits<StorageType>::digits, "Value does not fit in StorageType");
333  static_assert((MaskConstant<kRangeStart, kRangeLength>::value & MaskConstant<kValueStart, kValueLength>::value) == 0,
334  "Range and Value overlap");
335 
336  // Assert that SDK Part and Code fields fit in SdkCode field and don't overlap.
337  static_assert(kSdkPartStart + kSdkPartLength <= kValueLength, "SdkPart does not fit in Value");
338  static_assert(kSdkCodeStart + kSdkCodeLength <= kValueLength, "SdkCode does not fit in Value");
339  static_assert((MaskConstant<kSdkPartStart, kSdkPartLength>::value & MaskConstant<kSdkCodeStart, kSdkCodeLength>::value) == 0,
340  "SdkPart and SdkCode overlap");
341 
342  // Assert that Value fits in ValueType.
343  static_assert(kValueStart + kValueLength <= std::numeric_limits<ValueType>::digits, "Value does not fit in ValueType");
