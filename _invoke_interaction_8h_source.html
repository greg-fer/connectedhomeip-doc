27 /*
28  * A typed command invocation function that takes as input a cluster-object representation of a command request and
29  * callbacks for success and failure and either returns a decoded cluster-object representation of the response through
30  * the provided success callback or calls the provided failure callback.
31  *
32  * The RequestObjectT is generally expected to be a ClusterName::Commands::CommandName::Type struct, but any object
33  * that can be encoded using the DataModel::Encode machinery and exposes the GetClusterId() and GetCommandId() functions
34  * is expected to work.
35  *
36  * The ResponseObjectT is expected to be one of two things:
37  *
38  * - If a data response is expected on success, a struct type decodable via DataModel::Decode which has GetClusterId() and
39  * GetCommandId() methods. A ClusterName::Commands::ResponseCommandName::DecodableType is typically used.
40  * - If a status response is expected on success, a DataModel::NullObjectType.
41  *
42  */
43 template <typename ResponseObjectT = app::DataModel::NullObjectType, typename RequestObjectT>
44 CHIP_ERROR InvokeCommandRequest(Messaging::ExchangeManager * aExchangeMgr, SessionHandle sessionHandle, chip::EndpointId endpointId,
45  const RequestObjectT & requestCommandData,
46  typename TypedCommandCallback<ResponseObjectT>::OnSuccessCallbackType onSuccessCb,
47  typename TypedCommandCallback<ResponseObjectT>::OnErrorCallbackType onErrorCb)
48 {
49  app::CommandPathParams commandPath = { endpointId, 0, RequestObjectT::GetClusterId(), RequestObjectT::GetCommandId(),
50  (app::CommandPathFlags::kEndpointIdValid) };
51 
52  //
53  // Let's create a handle version of the decoder to ensure we do correct clean-up of it if things go south at any point below
54  //
55  auto decoder = chip::Platform::MakeUnique<TypedCommandCallback<ResponseObjectT>>(onSuccessCb, onErrorCb);
56  VerifyOrReturnError(decoder != nullptr, CHIP_ERROR_NO_MEMORY);
57 
58  //
59  // Upon successful completion of SendCommandRequest below, we're expected to free up the respective allocated objects
60  // in the OnDone callback.
61  //
62  auto onDone = [rawDecoderPtr = decoder.get()](app::CommandSender * commandSender) {
63  chip::Platform::Delete(commandSender);
64  chip::Platform::Delete(rawDecoderPtr);
65  };
66 
67  decoder->SetOnDoneCallback(onDone);
68 
69  auto commandSender = chip::Platform::MakeUnique<app::CommandSender>(decoder.get(), aExchangeMgr);
70  VerifyOrReturnError(commandSender != nullptr, CHIP_ERROR_NO_MEMORY);
71 
72  ReturnErrorOnFailure(commandSender->AddRequestData(commandPath, requestCommandData));
73  ReturnErrorOnFailure(commandSender->SendCommandRequest(sessionHandle));
74 
75  //
76  // We've effectively transfered ownership of the above allocated objects to CommandSender, and we need to wait for it to call us
77  // back when processing is completed (through OnDone) to eventually free up resources.
78  //
79  // So signal that by releasing the smart pointer.
80  //
81  decoder.release();
82  commandSender.release();
83 
84  return CHIP_NO_ERROR;
85 }
