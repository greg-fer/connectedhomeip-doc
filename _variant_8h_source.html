37 struct VariantCurry<Index, T, Ts...>

48  {
49  if (that_t == Index)

51  else
52  VariantCurry<Index + 1, Ts...>::Move(that_t, that_v, this_v);

56  {
57  if (that_t == Index)

59  new (this_v) T(*reinterpret_cast<const T *>(that_v));

63  VariantCurry<Index + 1, Ts...>::Copy(that_t, that_v, this_v);
64  }
