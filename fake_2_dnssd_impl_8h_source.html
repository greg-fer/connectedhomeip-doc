123  bool CheckMatch(CallType call, const DnssdService * service) const
124  {
125  bool callOk = call == callType;
126  bool instanceOk = strcmp(instanceName, service->mName) == 0;
127  bool hostOk = strcmp(hostName, service->mHostName) == 0;
128  bool serviceOk = strcmp(serviceName, service->mType) == 0;
129  bool subtypeNumOk = numSubtypes == service->mSubTypeSize;
130  bool txtNumOk = numTxt == service->mTextEntrySize;
131  if (!callOk || !instanceOk || !hostOk || !serviceOk || !subtypeNumOk || !txtNumOk)
132  {
133  return false;
134  }
135  // There aren't a lot of these, so just double loop for now.
136  // The orders of the txt and subtypes don't necessarily match.
137  for (size_t i = 0; i < service->mSubTypeSize; ++i)
138  {
139  if (!SubtypeIsExpected(service->mSubTypes[i]))

141  return false;
142  }
143  }
144  for (size_t i = 0; i < service->mTextEntrySize; ++i)
