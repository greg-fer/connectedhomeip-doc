29 /*
30  * An adapter callback that permits applications to provide std::function callbacks for success, error and on done.
31  * This permits a slightly more flexible programming model that allows applications to pass in lambdas and bound member functions
32  * as they see fit instead.
33  *
34  */

36 class WriteCallback final : public app::WriteClient::Callback
37 {
38 public:
39  using OnSuccessCallbackType = std::function<void(const app::ConcreteAttributePath &)>;
40 
41  //
42  // Callback to deliver any error that occurs during the write. This includes
43  // errors global to the write as a whole (e.g timeout) as well as per-attribute
44  // errors.
45  //
46  // In the latter case, path will be non-null. Otherwise, it shall be null.
47  //
48  using OnErrorCallbackType =
49  std::function<void(const app::ConcreteAttributePath * path, app::StatusIB status, CHIP_ERROR aError)>;
50  using OnDoneCallbackType = std::function<void(app::WriteClient *, WriteCallback *)>;
51 
52  WriteCallback(OnSuccessCallbackType aOnSuccess, OnErrorCallbackType aOnError, OnDoneCallbackType aOnDone) :
53  mOnSuccess(aOnSuccess), mOnError(aOnError), mOnDone(aOnDone)
54  {}
55 
56  void OnResponse(const app::WriteClient * apWriteClient, const app::ConcreteAttributePath & aPath, app::StatusIB status) override
57  {
58  if (status.mStatus == Protocols::InteractionModel::Status::Success)
59  {
60  mOnSuccess(aPath);
61  }
62  else
63  {
64  mOnError(&aPath, status, CHIP_ERROR_IM_STATUS_CODE_RECEIVED);
65  }
66  }
67 
68  void OnError(const app::WriteClient * apWriteClient, CHIP_ERROR aError) override
69  {
70  mOnError(nullptr, app::StatusIB(Protocols::InteractionModel::Status::Failure), aError);
71  }
72 
73  void OnDone(app::WriteClient * apWriteClient) override { mOnDone(apWriteClient, this); }
74 
75 private:
76  OnSuccessCallbackType mOnSuccess;
77  OnErrorCallbackType mOnError;
78  OnDoneCallbackType mOnDone;
79 };
